```ts
async function loadUserData() {
	try {
		const user = await fetchUser()
		const orders = await fetchOrders(user.id)
		console.log(user, orders)
	} catch (error) {
		console.error('Failed to load data:', error)
	} finally {
		console.log('Cleanup complete')
	}
}
```
