```ts
async function readConfig() {
	if (Math.random() > 0.5) throw new Error('Missing config')
	return { theme: 'dark' }
}

async function loadSettings() {
	try {
		const config = await readConfig()
		return config
	} catch (error) {
		return { theme: 'light', error }
	} finally {
		// cleanup
	}
}
```
